#include <iostream>
using namespace std;

void insert(int array[],int elem,int tam,int place)
{
  if(place<tam)
  {
    array[place]=elem;
    place++;
  }
}

void print(int *p,int n)
{
  for(int*pt = p ; pt<p+n;pt++)
    {
      cout<<*pt<<" ";
    }
}


int main() {
  int n=0;
  int tam=10;
  int *p;
  p=new int [tam];
  
  insert(p,5,tam,n);
  insert(p,10,tam,n);
  insert(p,15,tam,n);
  print(p,n);

  return 0;
